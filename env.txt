# Flint library
# Flint2 takes a while to build
git clone https://github.com/fredrik-johansson/flint2

./configure
make
# No test?
sudo make install
cd ..

# Python wrapper
git clone https://github.com/python-flint/python-flint
cd python-flint
python3 setup.py build_ext
